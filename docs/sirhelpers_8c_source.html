50bool _sir_validupdatedata(sir_update_config_data* data) {
51 if (_sir_validptr(data)) {
52 /* Either/or: this structure is passed to functions that either
53 * update levels or options, not both. */
54 bool have_levels = _sir_validptrnofail(data->levels);
55 bool have_opts = _sir_validptrnofail(data->opts);
56
57 if (have_levels || have_opts)
58 return have_levels ? _sir_validlevels(*data->levels) :
59 _sir_validopts(*data->opts);
60
61 _sir_seterror(_SIR_E_INVALID);
62 assert(!"invalid update config data");
63 }
64
65 return false;
66}

233int _sir_getchar(void) {
234#if defined(_WIN32)
235 return _getch();
236#else
237 struct termios cur = {0};
238 struct termios new = {0};
239
240 int get = tcgetattr(STDIN_FILENO, &cur);
241 if (0 != get) {
242 _sir_handleerr(errno);
243 return -1;
244 }
245
246 memcpy(&new, &cur, sizeof(struct termios));
247 new.c_lflag &= ~(ICANON | ECHO);
248
249 int set = tcsetattr(STDIN_FILENO, TCSANOW, &new);
250 if (0 != set) {
251 _sir_handleerr(errno);
252 return -1;
253 }
254
255 int ch = getchar();
256
257 set = tcsetattr(STDIN_FILENO, TCSANOW, &cur);
258 if (0 != set) {
259 _sir_handleerr(errno);
260 return -1;
261 }
262
263 return ch;
264#endif
265}
