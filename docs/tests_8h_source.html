47# define INIT_BASE(var, l_stdout, o_stdout, l_stderr, o_stderr, p_name, init) \
48 sirinit var = {0}; \
49 var.d_stdout.opts = (o_stdout) > 0 ? o_stdout : SIRO_DEFAULT; \
50 var.d_stdout.levels = l_stdout; \
51 var.d_stderr.opts = (o_stderr) > 0 ? o_stderr : SIRO_DEFAULT; \
52 var.d_stderr.levels = l_stderr; \
53 if (_sir_validstrnofail(p_name)) \
54 _sir_strncpy(var.name, SIR_MAXNAME, p_name, SIR_MAXNAME); \
55 bool var##_init = false; \
56 if (init) \
57 var##_init = sir_init(&var);
