336# define INIT_BASE(var, l_stdout, o_stdout, l_stderr, o_stderr, p_name, init) \
337 sirinit var = {0}; \
338 var.d_stdout.opts = (o_stdout) > 0 ? (o_stdout) : SIRO_DEFAULT; \
339 var.d_stdout.levels = (l_stdout); \
340 var.d_stderr.opts = (o_stderr) > 0 ? (o_stderr) : SIRO_DEFAULT; \
341 var.d_stderr.levels = (l_stderr); \
342 if (_sir_validstrnofail(p_name)) \
343 (void)_sir_strncpy(var.name, SIR_MAXNAME, (p_name), SIR_MAXNAME); \
344 bool var##_init = false; \
345 if (init) \
346 var##_init = sir_init(&var); \
347 else \
348 (void)var##_init
349
350# define INIT_N(var, l_stdout, o_stdout, l_stderr, o_stderr, name) \
351 INIT_BASE(var, l_stdout, o_stdout, l_stderr, o_stderr, name, true)

353# define INIT_SL(var, l_stdout, o_stdout, l_stderr, o_stderr, name) \
354 INIT_BASE(var, l_stdout, o_stdout, l_stderr, o_stderr, name, false)
