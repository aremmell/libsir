46#define INIT_BASE(var, l_stdout, o_stdout, l_stderr, o_stderr, p_name, init) \
47 sirinit var = {0}; \
48 var.d_stdout.opts = (o_stdout) > 0 ? o_stdout : SIRO_DEFAULT; \
49 var.d_stdout.levels = l_stdout; \
50 var.d_stderr.opts = (o_stderr) > 0 ? o_stderr : SIRO_DEFAULT; \
51 var.d_stderr.levels = l_stderr; \
52 if (_sir_validstrnofail(p_name)) \
53 _sir_strncpy(var.name, SIR_MAXNAME, p_name, SIR_MAXNAME); \
54 bool var##_init = false; \
55 if (init) \
56 var##_init = sir_init(&var);
