41bool _sir_validstyle(sir_textstyle style, uint32_t* pattr, uint32_t* pfg, uint32_t* pbg) {
42
43 uint32_t attr = (style & _SIRS_ATTR_MASK);
44 uint32_t fore = (style & _SIRS_FG_MASK);
45 uint32_t back = (style & _SIRS_BG_MASK);
46
47 bool attrvalid = attr <= SIRS_DIM;
48 bool fgvalid = fore <= SIRS_FG_WHITE;
49 bool bgvalid = back <= SIRS_BG_WHITE && !_SIRS_SAME_COLOR(fore, back);

62 if (attrvalid && fgvalid && bgvalid)
63 return true;
64
65 _sir_seterror(_SIR_E_TEXTSTYLE);
66 assert("!invalid text style");
67
68 return false;

187 static const size_t idx_attr_start = 0;

189
190 static const size_t idx_fg_start = 3;
191 static const size_t idx_fg_end = 19;
192
193 static const size_t idx_bg_start = 20;
194 static const size_t idx_bg_end = _sir_countof(sir_style_16color_map) - 1;
195
196 size_t low = 0;
197 size_t high = 0;
198
199 if (style <= SIRS_DIM) {
200 /* looking up an attribute */
201 low = idx_attr_start;
202 high = idx_attr_end;
203 } else if (style <= SIRS_FG_WHITE) {
204 /* looking up a foreground color */
205 low = idx_fg_start;
206 high = idx_fg_end;
207 } else {
208 /* looking up a background color */
209 low = idx_bg_start;
210 high = idx_bg_end;
211 }
212
213 _SIR_DECLARE_BIN_SEARCH(low, high);
214 _SIR_BEGIN_BIN_SEARCH();
215
216 if (sir_style_16color_map[_mid].from == style)
217 return sir_style_16color_map[_mid].to;
218
219 int comparison = sir_style_16color_map[_mid].from < style ? 1 : -1;
220
221 _SIR_ITERATE_BIN_SEARCH(comparison);
222 _SIR_END_BIN_SEARCH();
223
224 return _sir_getprivstyle(SIRS_FG_DEFAULT);
225}

243
244 if (privfg != 0)

246
247 if (privbg != 0)
248 snprintf(bgfmt, 7, ";%hu", privbg);
249
250 /* '\x1b[n;nnn;nnnm' */
251 snprintf(buf, size, "%s%hu%s%sm", SIR_BEGINSTYLE, privattr, fgfmt, bgfmt);

254 }
