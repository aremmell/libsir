69bool _sir_write_stdio(HANDLE console, const sirchar_t* message, size_t len) {
70
71 DWORD chars = (DWORD)len;
72 DWORD written = 0;
73
74 do {
75 DWORD pass = 0;
76
77 if (!WriteConsole(console, message + written, chars - written, &pass, NULL)) {
78 _sir_handlewin32err(GetLastError());

80 }

82 written += pass;
83 } while (written < chars);
