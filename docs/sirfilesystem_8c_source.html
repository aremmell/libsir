95bool _sir_getappfilename(char* restrict buffer, size_t size) {
96
97 if (!_sir_validptr(buffer))

99

101 _sir_seterror(_SIR_E_INVALID);
102 return false;
103 }
104
105#pragma message("TODO: we're going to have to change these to return int, in cases like readlink needs more buffer space, we can't just return false.")
106
107 bool retval = true;
108
109#if !defined(_WIN32)
110# if defined(__linux__)
111# if defined(__HAVE_UNISTD_READLINK__)
112 ssize_t read = readlink("/proc/self/exe", buffer, size);
113 if (-1 == read || read > (ssize_t)size) {
114 bool too_small = read > (ssize_t)size;
115 // TODO: if too_small == true: buffer is too small; change size to a pointer so we can store
116 // the necesssary size there.
117 _sir_handleerr(errno);
118 retval = false;
119 }
120# else
121# error "unable to resolve readlink(); see man readlink and its feature test macro requirements."
122# endif
123# elif defined(__FreeBSD__)
124 int mib[4] = { CTL_KERN, KERN_PROC, KERN_PROC_PATHNAME, -1 };
125 if (0 != sysctl(mib, 4, buffer, &size, NULL, 0)) {
126 _sir_handleerr(errno);
127 retval = false;
128 }
129# elif defined(__APPLE__)
130 uint32_t size32 = (uint32_t)size;
131 if (0 != _NSGetExecutablePath(buffer, &size32)) {
132 // TODO: buffer is too small; need size32 bytes
133 _sir_handleerr(errno);
134 retval = false;
135 }
136# else

138# endif
139#else // _WIN32
140 DWORD ret = GetModuleFileNameA(NULL, buffer, (DWORD)size);
141 DWORD le = GetLastError();
142 if (0 == ret || ERROR_INSUFFICIENT_BUFFER == le) {
143 bool too_small = ERROR_INSUFFICIENT_BUFFER == le;
144 // TODO: if too_small == true: buffer is too small; ret contains the chars stored in buffer.
145 _sir_handlewin32err(GetLastError());
146 retval = false;

150}
