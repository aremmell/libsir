82void __sir_handleerr(int code, const sirchar_t* func, const sirchar_t* file, uint32_t line) {
83 if (SIR_E_NOERROR != code) {
84 sirchar_t message[SIR_MAXERROR] = {0};
85 int finderr = 0;
86
87 errno = SIR_E_NOERROR;
88#if defined(__HAVE_XSI_STRERROR_R__)
89 _sir_selflog("%s: using XSI strerror_r\n", __func__);
90 finderr = strerror_r(code, message, SIR_MAXERROR);
91# if defined(__HAVE_XSI_STRERROR_R_ERRNO__)
92 _sir_selflog("%s: using XSI strerror_r for glibc < 2.13\n", __func__);
93 if (finderr == -1)
94 finderr = errno;
95# endif
96#elif defined(__HAVE_GNU_STRERROR_R__)
97 _sir_selflog("%s: using GNU strerror_r\n", __func__);
98 char* tmp = strerror_r(code, message, SIR_MAXERROR);
99 if (tmp != message)
100 _sir_strncpy(message, SIR_MAXERROR, tmp, SIR_MAXERROR);
101#elif defined(__HAVE_STRERROR_S__)
102 _sir_selflog("%s: using strerror_s\n", __func__);
103 finderr = (int)strerror_s(message, SIR_MAXERROR, code);
104#else
105 _sir_selflog("%s: using strerror\n", __func__);
106 char* tmp = strerror(code);
107 _sir_strncpy(message, SIR_MAXERROR, tmp, strnlen(tmp, SIR_MAXERROR));
108#endif
109 if (0 == finderr && _sir_validstrnofail(message)) {
110 __sir_setoserror(code, message, func, file, line);
111 } else {
112 /* Per my reading of the man pages, GNU strerror_r and normal strerror "can't fail";
113 * they simply return a string such as "Unknown nnn error" if unable to look up an
114 * error code.
115 */
116#if defined(__HAVE_XSI_STRERROR_R__)
117 _sir_selflog("%s: strerror_r failed! error: %d\n", __func__, finderr);
118#elif defined(__HAVE_STRERROR_S__)
119 _sir_selflog("%s: strerror_s failed! error: %d\n", __func__, finderr);
120#endif
121 }
122 }
123
124#if defined(SIR_ASSERT_ON_ERROR)
125 assert(_SIR_E_NOERROR == code);
126#endif
127}

159sirerror_t _sir_geterror(sirchar_t message[SIR_MAXERROR]) {
160 _sir_resetstr(message);
161
162 size_t low = 0;
163 size_t high = _sir_countof(sir_errors) - 1;
164
165 _SIR_DECLARE_BIN_SEARCH(low, high);
166 _SIR_BEGIN_BIN_SEARCH();
167
168 if (sir_errors[_mid].e == sir_te.lasterror) {
169 sirchar_t* final = NULL;
170 bool alloc = false;
171
172 if (_SIR_E_PLATFORM == sir_errors[_mid].e) {
173 final = (sirchar_t*)calloc(SIR_MAXERROR, sizeof(sirchar_t));
174
175 if (_sir_validptr(final)) {
176 alloc = true;
177 snprintf(final, SIR_MAXERROR, sir_errors[_mid].msg, sir_te.os_error,
178 (_sir_validstrnofail(sir_te.os_errmsg) ? sir_te.os_errmsg : SIR_UNKNOWN));
179 }
180 }
181 else {
182 final = (sirchar_t*)sir_errors[_mid].msg;
183 }
184
185 int fmtmsg = snprintf(message, SIR_MAXERROR, SIR_ERRORFORMAT, sir_te.loc.func,
186 sir_te.loc.file, sir_te.loc.line, final);
187
188 _SIR_UNUSED(fmtmsg);
189 assert(fmtmsg >= 0);
190
191 if (alloc)
192 _sir_safefree(final);
193
194 return sir_errors[_mid].e;
195 }
196
197 int comparison = sir_errors[_mid].e < sir_te.lasterror ? 1 : -1;
198
199 _SIR_ITERATE_BIN_SEARCH(comparison);
200 _SIR_END_BIN_SEARCH();
201
202 assert(false && sir_te.lasterror);
203 return _SIR_E_UNKNOWN;
204}
