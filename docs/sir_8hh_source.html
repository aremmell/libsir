68 public:

71 static const string_t format(const string_t& format, ...) {
72 if (format.empty()) return string_t();
73
74 va_list args, copy;
75 va_start(args, format);
76 va_copy(copy, args);
77
78 int needed = std::vsnprintf(NULL, 0, format.c_str(), copy);
79 if (needed <= 0) return string_t();
80
81 std::size_t newsize = static_cast<std::size_t>(needed + 1);
82 std::vector<sirchar_t> vec(newsize, '\0');
83
84 int formatted = std::vsnprintf(&vec[0], newsize, format.c_str(), args);
85 if (formatted <= 0) return string_t();
86
87 return &vec[0];
88 }
89 }; /* !class serializable */
