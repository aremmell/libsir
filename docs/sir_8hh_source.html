675 template<typename TFunc>
676 class buffer : public std::streambuf {
677 public:
678 using array_type = std::array<char_type, SIR_MAXMESSAGE>;
679
680 buffer() = delete;
681
682 explicit buffer(TFunc pfn) : _pfn(pfn) {
683 [[maybe_unused]]
684 bool ok = throw_on_policy<TPolicy>(pfn != nullptr && _arr != nullptr);
685 reset();
686 }
687
688 ~buffer() override = default;
689
690 protected:
691 void reset() {
692 if (_arr) {
693 _arr->fill('\0');
694 setp(_arr->data(), _arr->data() + _arr->size() - 1);
695 }
696 }
697
698 bool write_out() {
699 for (auto it = _arr->rbegin(); it != _arr->rend(); it++) {
700 if ((*it != '\0') && (*it == '\n')) {
701 *it = '\0';
702 break;
703 }
704 }
705
706 const bool write = _pfn ? _pfn(_arr->data()) : false;
707 reset();
708 return throw_on_policy<TPolicy>(write);
709 }
710
711 int_type overflow(int_type ch) override {
712 if (ch != traits_type::eof() && write_out()) {
713 return ch;
714 }
715 return traits_type::eof();
716 }
717
718 int sync() override {
719 return write_out() ? 0 : -1;
720 }
721
722 std::streamsize xsputn(const char_type* s, std::streamsize count) override {
723 std::streamsize written = 0;
724 do {
725 ptrdiff_t left = epptr() - pptr();
726 if (!left) {
727 int_type ch = overflow(traits_type::to_int_type(*(s + written)));
728 if (ch != traits_type::eof()) {
729 continue;
730 } else {
731 break;
732 }
733 }
734
735 auto this_write = std::min(static_cast<std::streamsize>(left),
736 count - written);
737 std::memcpy(pptr(), s + written, static_cast<std::size_t>(this_write));
738 pbump(static_cast<int>(this_write));
739 written += this_write;
740 } while (written < count);
741
742 return written;
743 }
744
745 private:
746 std::shared_ptr<array_type> _arr = std::make_shared<array_type>();
747 TFunc _pfn = nullptr;
748 };
