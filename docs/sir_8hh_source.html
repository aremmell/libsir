252 template<DerivedFromPolicy TPolicy>
253 static inline bool throw_on_policy(bool expr) noexcept(false) {
254 if (!expr) {
255 SIR_ASSERT(expr);
256 if constexpr(TPolicy::throw_on_error()) {
257 throw exception::from_libsir_error();
258 }
259 }
260 return expr;
261 }

582 template<typename TFunc>
583 class buffer : public std::streambuf {
584 public:
585 using array_type = std::array<char_type, SIR_MAXMESSAGE>;
586
587 buffer() = delete;
588
589 explicit buffer(TFunc pfn) : _pfn(pfn) {
590 [[maybe_unused]]
591 bool ok = throw_on_policy<TPolicy>(pfn != nullptr && _arr != nullptr);
592 reset();
593 }
594
595 ~buffer() override = default;
596
597 protected:
598 void reset() {
599 if (_arr) {
600 _arr->fill('\0');
601 setp(_arr->data(), _arr->data() + _arr->size() - 1);
602 }
603 }
604
605 bool write_out() {
606 for (auto it = _arr->rbegin(); it != _arr->rend(); it++) {
607 if ((*it != '\0') && (*it == '\n')) {
608 *it = '\0';
609 break;
610 }
611 }
612
613 const bool write = _pfn ? _pfn(_arr->data()) : false;
614 reset();
615 return throw_on_policy<TPolicy>(write);
616 }
617
618 int_type overflow(int_type ch) override {
619 if (ch != traits_type::eof() && write_out()) {
620 return ch;
621 }
622 return traits_type::eof();
623 }
624
625 int sync() override {
626 return write_out() ? 0 : -1;
627 }
628
629 std::streamsize xsputn(const char_type* s, std::streamsize count) override {
630 std::streamsize written = 0;
631 do {
632 ptrdiff_t left = epptr() - pptr();
633 if (!left) {
634 int_type ch = overflow(traits_type::to_int_type(*(s + written)));
635 if (ch != traits_type::eof()) {
636 continue;
637 } else {
638 break;
639 }
640 }
641
642 auto this_write = std::min(static_cast<std::streamsize>(left),
643 count - written);
644 memcpy(pptr(), s + written, static_cast<size_t>(this_write));
645 pbump(static_cast<int>(this_write));
646 written += this_write;
647 } while (written < count);
648
649 return written;
650 }
651
652 private:
653 std::shared_ptr<array_type> _arr = std::make_shared<array_type>();
654 TFunc _pfn = nullptr;
655 };

683 template<size_t N, typename ...Ts>
684 using NthTypeOf = typename std::tuple_element<N, std::tuple<Ts...>>::type;
685
687 template<size_t N, typename TBase, typename ...Ts>
688 consteval auto all_derived_from_t() {
689 if constexpr(!std::is_base_of_v<TBase, NthTypeOf<N, Ts...>>)
690 return std::false_type::value;
691 if constexpr(N == 0)
692 return std::true_type::value;
693 else
694 return all_derived_from_t<N - 1, TBase, Ts...>();
695 }

699 concept DerivedFromT = all_derived_from_t<sizeof...(Ts) - 1, TBase, Ts...>();
