172static bool _sirmutex_waitwin32(sirmutex_t mutex, DWORD msec) {
173
174 if (_sir_validptr(mutex)) {
175 DWORD wait = WaitForSingleObject(mutex, msec);
176
177 switch (wait) {
178 case WAIT_ABANDONED:
179 case WAIT_FAILED:
180 _sir_selflog(
