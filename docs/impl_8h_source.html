48/*
49 * Appends src to string dst of size dsize (unlike strncat, dsize is the
50 * full size of dst, not space left). At most dsize-1 characters
51 * will be copied. Always NUL terminates (unless dsize <= strlen(dst)).
52 * Returns strlen(src) + MIN(dsize, strlen(initial dst)).
53 * If retval >= dsize, truncation occurred.
54 */
55
56static inline size_t
57_sir_strlcat(char *dst, const char *src, size_t dsize)
58{
59 const char *odst = dst;
60 const char *osrc = src;
61 size_t n = dsize;
62 size_t dlen;
63
64 /*
65 * Find the end of dst and adjust bytes
66 * left, but don't go past the end.
67 */
68
69 while (n-- != 0 && *dst != '\0')
70 dst++;
71
72 dlen = dst - odst;
73 n = dsize - dlen;
74
75 if (n-- == 0)
76 return dlen + strlen(src);
77
78 while (*src != '\0') {
79 if (n != 0) {
80 *dst++ = *src;
81 n--;
82 }
83 src++;
84 }
85
86 *dst = '\0';
87
88 return dlen + (src - osrc); /* count does not include NUL */
89}
90
91# define strlcat _sir_strlcat
92# define SIR_IMPL_STRLCAT_DEF 1
93#endif /* SIR_IMPL_STRLCAT */

105/*
106 * Copy string src to buffer dst of size dsize. At most dsize-1
107 * chars will be copied. Always NUL terminates (unless dsize == 0).
108 * Returns strlen(src); if retval >= dsize, truncation occurred.
109 */
110
111static inline size_t
112_sir_strlcpy(char *dst, const char *src, size_t dsize)
113{
114 const char *osrc = src;
115 size_t nleft = dsize;
116
117 /*
118 * Copy as many bytes as will fit.
119 */
120
121 if (nleft != 0) {
122 while (--nleft != 0) {
123 if ((*dst++ = *src++) == '\0')
124 break;
125 }
126 }
127
128 /*
129 * Not enough room in dst, add
130 * NUL and traverse rest of src.
131 */
132
133 if (nleft == 0) {
134 if (dsize != 0)
135 *dst = '\0'; /* NUL-terminate dst */
136 while (*src++);
137 }
138
139 return src - osrc - 1; /* count does not include NUL */
140}
141
142# define strlcpy _sir_strlcpy
143# define SIR_IMPL_STRLCPY_DEF 1
144#endif /* SIR_IMPL_STRLCPY */

156static inline size_t
157_sir_strnlen(const char *str, size_t maxlen)
158{
159 const char *cp;
160
161 for (cp = str; maxlen != 0 && *cp != '\0'; cp++, maxlen--);
162
163 return (size_t)(cp - str);
164}
165

167# define SIR_IMPL_STRNLEN_DEF 1
168#endif /* SIR_IMPL_STRNLEN */

179static inline char*
180_sir_strndup(const char *str, size_t maxlen)
181{
182 char *copy;
183 size_t len;
184
185 len = _sir_strnlen(str, maxlen);
186
187 copy = (char*)malloc(len + 1);
188 if (copy != NULL) {
189 (void)memcpy(copy, str, len);
190 copy[len] = '\0';
191 }
192
193 return copy;
194}
