254bool _sir_mapmutexid(sir_mutex_id mid, sirmutex_t** m, void** section) {
255
256 if (!_sir_validptr(m))
257 return false;
258
259 sirmutex_t* tmpm;
260 void* tmpsec;
261
262 switch (mid) {
263 case _SIRM_INIT:
264 _sir_once(&si_once, _sir_initmutex_si_once);
265 tmpm = &si_mutex;
266 tmpsec = &_sir_si;
267 break;
268 case _SIRM_FILECACHE:
269 _sir_once(&fc_once, _sir_initmutex_fc_once);
270 tmpm = &fc_mutex;
271 tmpsec = &_sir_fc;
272 break;
273 case _SIRM_TEXTSTYLE:
274 _sir_once(&ts_once, _sir_initmutex_ts_once);
275 tmpm = &ts_mutex;
276 tmpsec = sir_override_style_map;
277 break;
278 default: tmpm = NULL; tmpsec = NULL;
279 }
280
281 *m = tmpm;
282 if (section)
283 *section = tmpsec;
284
285 return *m != NULL && (!section || *section != NULL);
286}

288bool _sir_cleanup(void) {
289
290 if (!_sir_sanity())
291 return false;
292
293 bool cleanup = true;
294 sirfcache* sfc = _sir_locksection(_SIRM_FILECACHE);
295 assert(sfc);
296
297 if (_sir_notnull(sfc)) {
298 bool destroyfc = _sir_fcache_destroy(sfc);
299 assert(destroyfc);
300 cleanup &= _sir_unlocksection(_SIRM_FILECACHE) && destroyfc;
301 }
302
303 sirinit* si = _sir_locksection(_SIRM_INIT);
304 assert(si);
305 cleanup &= _sir_notnull(si);
306
307 if (_sir_notnull(si)) {
308 _sir_syslog_close(&si->d_syslog);
309 _sir_resettextstyles();
310
311#if !defined(_WIN32)
312 atomic_store(&_sir_magic, 0);
313#else
314 _sir_magic = 0;
315#endif
316 memset(si, 0, sizeof(sirinit));
317 cleanup &= _sir_unlocksection(_SIRM_INIT);
318 }
319
320 _sir_selflog("%s: libsir unloaded cleanly: %s\n", __func__, (cleanup ? "yes" : "NO!"));
321
322 assert(cleanup);
323 return cleanup;
324}

391bool _sir_logv(sir_level level, const sirchar_t* format, va_list args) {
392
393 _sir_seterror(_SIR_E_NOERROR);
394
395 if (!_sir_sanity() || !_sir_validlevel(level) || !_sir_validstr(format))
396 return false;
397
398 sirinit* si = _sir_locksection(_SIRM_INIT);
399 if (!si)
400 return false;
401
402 sirinit tmpsi = {0};
403 memcpy(&tmpsi, si, sizeof(sirinit));
404
405 _sir_unlocksection(_SIRM_INIT);
406
407 sirbuf buf = {0};
408
409 bool appliedstyle = false;
410 sir_textstyle style = _sir_gettextstyle(level);
411 assert(SIRS_INVALID != style);
412
413 if (SIRS_INVALID != style) {
414 appliedstyle = _sir_formatstyle(style, buf.style, SIR_MAXSTYLE);
415 assert(appliedstyle);
416 }
417
418 time_t now;
419 long nowmsec;
420 bool gettime = _sir_getlocaltime(&now, &nowmsec);
421 assert(gettime);
422
423 if (gettime) {
424 bool fmttime = _sir_formattime(now, buf.timestamp, SIR_TIMEFORMAT);
425 _SIR_UNUSED(fmttime);
426
427 if (0 > snprintf(buf.msec, SIR_MAXMSEC, SIR_MSECFORMAT, nowmsec))
428 _sir_handleerr(errno);
429 }
430
431 if (0 > snprintf(buf.level, SIR_MAXLEVEL, SIR_LEVELFORMAT, _sir_levelstr(level)))
432 _sir_handleerr(errno);
433
434 if (_sir_validstrnofail(tmpsi.processName))
435 _sir_strncpy(buf.name, SIR_MAXNAME, tmpsi.processName, SIR_MAXNAME);
436
437 pid_t pid = _sir_getpid();
438 if (0 > snprintf(buf.pid, SIR_MAXPID, SIR_PIDFORMAT, pid))
439 _sir_handleerr(errno);
440
441 pid_t tid = _sir_gettid();
442 if (tid != pid) {
443 if (!_sir_getthreadname(buf.tid)) {
444 if (0 > snprintf(buf.tid, SIR_MAXPID, SIR_PIDFORMAT, tid))
445 _sir_handleerr(errno);
446 }
447 }
448
449 if (0 > vsnprintf(buf.message, SIR_MAXMESSAGE, format, args))
450 _sir_handleerr(errno);
451
452 return _sir_dispatch(&tmpsi, level, &buf);
453}

455bool _sir_dispatch(sirinit* si, sir_level level, sirbuf* buf) {
456
457 bool retval = true;
458 size_t dispatched = 0;
459 size_t wanted = 0;
460
461 if (_sir_bittest(si->d_stdout.levels, level)) {
462 const sirchar_t* write = _sir_format(true, si->d_stdout.opts, buf);
463 bool wrote = _sir_validstrnofail(write) &&
464 _sir_write_stdout(write, buf->output_len);
465 retval &= wrote;
466
467 if (wrote)
468 dispatched++;
469 wanted++;
470 }
471
472 if (_sir_bittest(si->d_stderr.levels, level)) {
473 const sirchar_t* write = _sir_format(true, si->d_stderr.opts, buf);
474 bool wrote = _sir_validstrnofail(write) &&
475 _sir_write_stderr(write, buf->output_len);
476 retval &= wrote;
477
478 if (wrote)
479 dispatched++;
480 wanted++;
481 }
482
483 if (_sir_bittest(si->d_syslog.levels, level)) {
484 if (_sir_syslog_write(level, buf, &si->d_syslog))
485 dispatched++;
486 wanted++;
487 }
488
489 sirfcache* sfc = _sir_locksection(_SIRM_FILECACHE);
490 if (sfc) {
491 size_t fdispatched = 0;
492 size_t fwanted = 0;
493 retval &= _sir_fcache_dispatch(sfc, level, buf, &fdispatched, &fwanted);
494 retval &= _sir_unlocksection(_SIRM_FILECACHE);
495 dispatched += fdispatched;
496 wanted += fwanted;
497 }
498
499 if (0 == wanted) {
500 _sir_seterror(_SIR_E_NODEST);
501 _sir_selflog("%s: no destinations registered to for level %04x\n", __func__, level);
502 return false;
503 }
504
505 return retval && (dispatched == wanted);
506}

508const sirchar_t* _sir_format(bool styling, sir_options opts, sirbuf* buf) {
509
510 if (_sir_validopts(opts) && _sir_validptr(buf) && _sir_validptr(buf->output)) {
511 bool first = true;
512
513 _sir_resetstr(buf->output);
514
515 if (styling)
516 _sir_strncat(buf->output, SIR_MAXOUTPUT, buf->style, SIR_MAXSTYLE);
517
518 if (!_sir_bittest(opts, SIRO_NOTIME)) {
519 _sir_strncat(buf->output, SIR_MAXOUTPUT, buf->timestamp, SIR_MAXTIME);
520 first = false;
521
522#if defined(SIR_MSEC_TIMER)
523 if (!_sir_bittest(opts, SIRO_NOMSEC))
524 _sir_strncat(buf->output, SIR_MAXOUTPUT, buf->msec, SIR_MAXMSEC);
525#endif
526 }
527
528 if (!_sir_bittest(opts, SIRO_NOLEVEL)) {
529 if (!first)
530 _sir_strncat(buf->output, SIR_MAXOUTPUT, " ", 1);
531 _sir_strncat(buf->output, SIR_MAXOUTPUT, buf->level, SIR_MAXLEVEL);
532 first = false;
533 }
534
535 bool name = false;
536 if (!_sir_bittest(opts, SIRO_NONAME) && _sir_validstrnofail(buf->name)) {
537 if (!first)
538 _sir_strncat(buf->output, SIR_MAXOUTPUT, " ", 1);
539 _sir_strncat(buf->output, SIR_MAXOUTPUT, buf->name, SIR_MAXNAME);
540 first = false;
541 name = true;
542 }
543
544 bool wantpid = !_sir_bittest(opts, SIRO_NOPID) && _sir_validstrnofail(buf->pid);
545 bool wanttid = !_sir_bittest(opts, SIRO_NOTID) && _sir_validstrnofail(buf->tid);
546
547 if (wantpid || wanttid) {
548 if (name)
549 _sir_strncat(buf->output, SIR_MAXOUTPUT, "(", 1);
550 else if (!first)
551 _sir_strncat(buf->output, SIR_MAXOUTPUT, " ", 1);
552
553 if (wantpid)
554 _sir_strncat(buf->output, SIR_MAXOUTPUT, buf->pid, SIR_MAXPID);
555
556 if (wanttid) {
557 if (wantpid)
558 _sir_strncat(buf->output, SIR_MAXOUTPUT, SIR_PIDSEPARATOR, 1);
559 _sir_strncat(buf->output, SIR_MAXOUTPUT, buf->tid, SIR_MAXPID);
560 }
561
562 if (name)
563 _sir_strncat(buf->output, SIR_MAXOUTPUT, ")", 1);
564 }
565
566 if (!first)
567 _sir_strncat(buf->output, SIR_MAXOUTPUT, ": ", 2);
568
569 _sir_strncat(buf->output, SIR_MAXOUTPUT, buf->message, SIR_MAXMESSAGE);
570
571 if (styling)
572 _sir_strncat(buf->output, SIR_MAXOUTPUT, SIR_ENDSTYLE, SIR_MAXSTYLE);
573
574 _sir_strncat(buf->output, SIR_MAXOUTPUT, "\n", 1);
575
576 buf->output_len = strnlen(buf->output, SIR_MAXOUTPUT);
577
578 return buf->output;
579 }
580
581 return NULL;
582}

596void _sir_syslog_open(const char* app_name, sir_syslog_dest* ctx) {
597 if (!ctx->opened_log) {
598 char identity[SIR_MAXPATH] = {0};
599
600 if (_sir_validstrnofail(app_name)) {
601 /* using app name from config. */
602 if (SIR_USING_SYSLOG) {
603 _sir_strncpy(identity, SIR_MAXPATH, app_name, strnlen(app_name, SIR_MAXNAME));
604 } else {
605 snprintf(identity, SIR_MAXPATH, "com.%s.%s", app_name, SIR_LIBNAME);
606 }
607 } else {
608 /* no name in config; using name of binary */
609 char appfilename[SIR_MAXPATH] = {0};
610 if (_sir_getappfilename(appfilename, SIR_MAXPATH)) {
611 if (SIR_USING_SYSLOG) {
612 _sir_strncpy(identity, SIR_MAXPATH, appfilename, strnlen(appfilename, SIR_MAXPATH));
613 } else {
614 snprintf(identity, SIR_MAXPATH, "com.%s.%s", appfilename, SIR_LIBNAME);
615 }
616 } else {
617 /* everything failed; using 'libsir' */
618 _sir_strncpy(identity, SIR_MAXPATH, SIR_LIBNAME, strnlen(SIR_LIBNAME, SIR_MAXPATH));
619 }
620 }
621
622#if defined(__APPLE__)
623
624# pragma message("TODO: add these fields to sirinit. user might want to use something other than the process name for ident")
625 const char* category = "minutiae";
626
627 ctx->logger = os_log_create(identity, category);
628 ctx->opened_log = true;
629 _sir_selflog("%s: opened os_log ('%s', '%s')\n", __func__, identity, category);
630
631#else
632 int logopt = LOG_NDELAY | (ctx->include_pid ? LOG_PID : 0);
633 int facility = LOG_USER;
634
635 openlog(identity, logopt, facility);
636 ctx->opened_log = true;
637 _sir_selflog("%s: opened syslog(%s, %x, %x)\n", __func__, identity, logopt, facility);
638#endif
639 } else {
640 _sir_selflog("%s: already opened log; ignoring\n", __func__);
641 }
642}

749bool _sir_getlocaltime(time_t* tbuf, long* nsecbuf) {
750 if (tbuf) {
751 time(tbuf);
752#if defined(SIR_MSEC_POSIX)
753 struct timespec ts = {0};
754 int clock = clock_gettime(SIR_MSECCLOCK, &ts);
755 assert(0 == clock);
756
757 if (0 == clock) {
758 if (nsecbuf) {
759 *nsecbuf = (ts.tv_nsec / 1e6);
760 assert(*nsecbuf < 1000);
761 }
762 } else {
763 *nsecbuf = 0;
764 _sir_selflog("%s: clock_gettime failed; errno: %d\n", __func__, errno);
765 }
766#elif defined(SIR_MSEC_WIN32)
767 static const ULONGLONG uepoch = (ULONGLONG)116444736e9;
768
769 FILETIME ftutc = {0};
770 GetSystemTimePreciseAsFileTime(&ftutc);
771
772 ULARGE_INTEGER ftnow;
773 ftnow.HighPart = ftutc.dwHighDateTime;
774 ftnow.LowPart = ftutc.dwLowDateTime;
775 ftnow.QuadPart = (ULONGLONG)((ftnow.QuadPart - uepoch) / 1e7);
776
777 *tbuf = (time_t)ftnow.QuadPart;
778
779 if (nsecbuf) {
780 SYSTEMTIME st = {0};
781 FileTimeToSystemTime(&ftutc, &st);
782 *nsecbuf = st.wMilliseconds;
783 }
784#else
785 time(tbuf);
786 if (nsecbuf)
787 *nsecbuf = 0;
788#endif
789 return true;
790 }
791
792 return false;
793}
