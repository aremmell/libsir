66# define _SIR_DECLARE_BIN_SEARCH(low, high) \
67 size_t _low = low; \
68 size_t _high = high; \
69 size_t _mid = (_low + _high) / 2;
70
71# define _SIR_BEGIN_BIN_SEARCH() do {
72# define _SIR_ITERATE_BIN_SEARCH(comparison) \
73 if (0 == comparison) { \
74 break; \
75 } \
76 \
77 if (_low == _high) \
78 break; \
79 \
80 if (0 > comparison && (_mid - 1) >= _low) { \
81 _high = _mid - 1; \
82 } else if ((_mid + 1) <= _high) { \
83 _low = _mid + 1; \
84 } else { \
85 break; \
86 } \
87 \
88 _mid = (_low + _high) / 2;
89# define _SIR_END_BIN_SEARCH() \
90 } while (true);

312uint64_t FNV_1a(const char* str)
313{
314 uint64_t hash = 14695981039346656037UL;
315 for (const char* c = str; *c; c++) {
316 hash ^= (uint64_t)(unsigned char)(*c);
317 hash *= 1099511628211UL;
318 }
