378static inline
379struct tm* _sir_localtime(const time_t* timer, struct tm* buf) {
380 if (!timer || !buf)
381 return NULL;
382# if defined(__HAVE_STDC_SECURE_OR_EXT1__) && !defined(__EMBARCADEROC__)
383# if !defined(__WIN__)
384 struct tm* ret = localtime_s(timer, buf);
385 if (!ret) {
386 (void)_sir_handleerr(errno);
387 return NULL;
388 }
389# else /* __WIN__ */
390 errno_t ret = localtime_s(buf, timer);
391 if (0 != ret) {
392 (void)_sir_handleerr(ret);
393 return NULL;
394 }
395# endif
396 return buf;
397# else /* !__HAVE_STDC_SECURE_OR_EXT1__ */
398# if !defined(__WIN__) || defined(__EMBARCADEROC__)
399 struct tm* ret = localtime_r(timer, buf);
400# else
401 struct tm* ret = localtime(timer);
402# endif
403 if (!ret)
404 (void)_sir_handleerr(errno);
405 return ret;
406# endif
407}

449static inline
450uint32_t FNV32_1a(const uint8_t* data, size_t len) {
451 uint32_t hash = 2166136261U;
452 for (size_t n = 0; n < len; n++) {
453 hash ^= (uint32_t)data[n];
454 hash = (uint32_t)(hash * 16777619ULL);
455 }
456 return hash;
457}
