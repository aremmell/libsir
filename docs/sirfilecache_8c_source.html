167bool _sirfile_write(sirfile* sf, const sirchar_t* output) {
168
169 if (_sirfile_validate(sf) && _sir_validstr(output)) {
170
171 if (_sirfile_needsroll(sf)) {
172 bool rolled = false;
173 sirchar_t* newpath = NULL;
174
175 if (_sirfile_roll(sf, &newpath)) {
176 sirchar_t header[SIR_MAXMESSAGE] = {0};
177 snprintf(header, SIR_MAXMESSAGE, SIR_FHROLLED, newpath);
178 rolled = _sirfile_writeheader(sf, header);
179 }
180
181 _sir_safefree(newpath);
182 if (!rolled)
183 return false;
184 }
185
186 size_t writeLen = strnlen(output, SIR_MAXOUTPUT);
187 size_t write = fwrite(output, sizeof(sirchar_t), writeLen, sf->f);
188
189 assert(write == writeLen);
190
191 if (write < writeLen) {
192 int err = ferror(sf->f);
193 int eof = feof(sf->f);
194
195 _sir_handleerr(err);
196 _sir_handleerr(eof);
197
198 _sir_selflog("%s: wrote %lu/%lu bytes to %d; ferror: %d, feof: %d\n", __func__, write, writeLen,
199 sf->id, err, eof);
200
206 clearerr(sf->f);
207 }
208
209 return write == writeLen;
210 }
211
212 return false;

395sirfileid_t _sir_fcache_add(sirfcache* sfc, const sirchar_t* path, sir_levels levels, sir_options opts) {
396
397 if (_sir_validptr(sfc) && _sir_validstr(path) && _sir_validlevels(levels) && _sir_validopts(opts)) {
398
399 if (sfc->count >= SIR_MAXFILES) {
400 _sir_seterror(_SIR_E_FCFULL);
401 return NULL;
402 }
403
404 sirfile* existing = _sir_fcache_find(sfc, (const void*)path, _sir_fcache_pred_path);
405
406 if (NULL != existing) {
407 _sir_seterror(_SIR_E_DUPFILE);
408 _sir_selflog("%s: file with path '%s' already added.\n", __func__, path);
409 return NULL;
410 }
411
412 sirfile* sf = _sirfile_create(path, levels, opts);
413 if (_sirfile_validate(sf)) {
414 sfc->files[sfc->count++] = sf;
415
416 if (!_sir_bittest(sf->opts, SIRO_NOHDR))
417 _sirfile_writeheader(sf, SIR_FHBEGIN);
418
419 return &sf->id;
420 }
421 }

514bool _sir_fcache_dispatch(sirfcache* sfc, sir_level level, sirbuf* buf,
515 size_t* dispatched, size_t* wanted) {
516
517 if (_sir_validptr(sfc) && _sir_validlevel(level) && _sir_validptr(buf)
518 && _sir_validptr(dispatched) && _sir_validptr(wanted)) {
519
520 bool r = true;
521 const sirchar_t* write = NULL;
522 sir_options lastopts = 0;
523
524 *dispatched = 0;
525 *wanted = 0;
526
527 for (size_t n = 0; n < sfc->count; n++) {
528 assert(_sirfile_validate(sfc->files[n]));
529
530 if (!_sir_bittest(sfc->files[n]->levels, level)) {
531 _sir_selflog("%s: levels for file %d (%04lx) don't include (%04lx); skipping...\n", __func__,
532 sfc->files[n]->id, sfc->files[n]->levels, level);
533 continue;
534 }
535
536 (*wanted)++;
537
538 if (!write || sfc->files[n]->opts != lastopts) {
539 write = _sir_format(false, sfc->files[n]->opts, buf);
540 assert(write);
541 lastopts = sfc->files[n]->opts;
542 }
543
544 if (write && _sirfile_write(sfc->files[n], write)) {
545 r &= true;
546 (*dispatched)++;
547 } else {
548 _sir_selflog("%s: write to %d failed!\n", __func__, sfc->files[n]->id);
549 }
550 }
551
552 if (*dispatched > 0)
553 _sir_fflush_all();
554
555 return r && (*dispatched == *wanted);
556 }
557
558 return false;
559}
