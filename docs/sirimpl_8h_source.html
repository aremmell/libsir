27/* OpenBSD strlcat() */
28
29/*
30 * Appends src to string dst of size dsize (unlike strncat, dsize is the
31 * full size of dst, not space left). At most dsize-1 characters
32 * will be copied. Always NUL terminates (unless dsize <= strlen(dst)).
33 * Returns strlen(src) + MIN(dsize, strlen(initial dst)).
34 * If retval >= dsize, truncation occurred.
35 */
36
37static inline size_t
38_sir_strlcat(char *dst, const char *src, size_t dsize)
39{
40 const char * odst = dst;
41 const char * osrc = src;
42 size_t n = dsize;
43 size_t dlen;
44
45 /* Find the end of dst and adjust bytes left but don't go past end. */
46
47 while (n-- != 0 && *dst != '\0')
48 dst++;
49
50 dlen = dst - odst;
51 n = dsize - dlen;
52
53 if (n-- == 0)
54 return dlen + strlen(src);
55
56 while (*src != '\0') {
57 if (n != 0) {
58 *dst++ = *src;
59 n--;
60 }
61 src++;
62 }
63 *dst = '\0';
64
65 return dlen + (src - osrc); /* count does not include NUL */
66}
67# define strlcat _sir_strlcat
68# endif /* SIR_IMPL_STRLCAT */

73/* OpenBSD strlcpy */
74
75/*
76 * Copy string src to buffer dst of size dsize. At most dsize-1
77 * chars will be copied. Always NUL terminates (unless dsize == 0).
78 * Returns strlen(src); if retval >= dsize, truncation occurred.
79 */
80
81static inline size_t
82_sir_strlcpy(char *dst, const char *src, size_t dsize)
83{
84 const char * osrc = src;
85 size_t nleft = dsize;
86
87 /* Copy as many bytes as will fit. */
88 if (nleft != 0) {
89 while (--nleft != 0) {
90 if ((*dst++ = *src++) == '\0')
91 break;
92 }
93 }
94
95 /* Not enough room in dst, add NUL and traverse rest of src. */
96 if (nleft == 0) {
97 if (dsize != 0)
98 *dst = '\0'; /* NUL-terminate dst */
99 while (*src++);
100 }
101
102 return src - osrc - 1; /* count does not include NUL */
103}
104# define strlcpy _sir_strlcpy
105# endif /* SIR_IMPL_STRLCPY */
