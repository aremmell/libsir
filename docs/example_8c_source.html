50 _SIR_UNUSED(argc);

53 /*
54 * Instantiate the initialization structure.
55 *
56 * NOTE: It is not necessary to retain this structure in memory;
57 * libsir makes a copy of it before returning from sir_init.
58 */
59 sirinit si = {0};

61 /* Configure levels for stdout. Send debug, information, warning, and notice messages there. */
62 si.d_stdout.levels = SIRL_DEBUG | SIRL_INFO | SIRL_WARN | SIRL_NOTICE;
63
64 /* Configure options for stdout. Don't show the time stamp, process ID, or thread ID. */
65 si.d_stdout.opts = SIRO_NOTIME | SIRO_NOPID | SIRO_NOTID;
66
67 /* Configure levels for stderr. Send error and above there. */
68 si.d_stderr.levels = SIRL_ERROR | SIRL_CRIT | SIRL_ALERT | SIRL_EMERG;
69
70 /* Configure options for stderr. Don't show the process ID or thread ID. */
71 si.d_stderr.opts = SIRO_NOPID | SIRO_NOTID;
72
73 /* Configure options for syslog. Don't send any output there. */
74 si.d_syslog.levels = SIRL_NONE;
75
76 /* Configure a name to associate with our output. */
77 static const char* appname = "myapp";
78 _sir_strncpy(si.processName, SIR_MAXNAME, appname, strnlen(appname, SIR_MAXNAME));
79
80 /* Initialize libsir. */
81 if (!sir_init(&si))
82 return report_error();
83
84 /* Configure and add a log file; don't log the process name, and send all levels there. */
85 static const sirchar_t* log_file_name = "libsir-example.log";
86 sirfileid_t fileid1 = sir_addfile(log_file_name, SIRL_ALL, SIRO_NONAME);
87 if (NULL == fileid1)
88 return report_error();
89
90
91 /* Now we're ready to start generating output. */
92 int n = 12345;
93 const char* somestr = "some string that's useful";
94 float f = 0.09f;
95
96 /*
97 * This will be sent to all destinations registered for ::SIRL_DEBUG.
98 * Notice that it is not necessary to add a newline at the end. libsir does
99 * this automatically.
100 */
101 sir_debug("some debug data: {n=%d, somestr='%s', f=%.04f}", n, somestr, f);
102
103 /* Do the same for the rest of available severity levels. */
104 sir_info("some information: {n=%d, somestr='%s', f=%.04f}", n, somestr, f);
105
106 sir_notice("consider yourself on notice: {n=%d, somestr='%s', f=%.04f}", n, somestr, f);
107
108 sir_warn("a stern warning {n=%d, somestr='%s', f=%.04f}", n, somestr, f);
109
110 sir_error("uh-oh, an error has occurred {n=%d, somestr='%s', f=%.04f}", n, somestr, f);
111
112 sir_crit("we've reached critical mass {n=%d, somestr='%s', f=%.04f}", n, somestr, f);
113
114 sir_alert("sound the alarm! {n=%d, somestr='%s', f=%.04f}", n, somestr, f);
115
116 sir_emerg("somebody call 911 {n=%d, somestr='%s', f=%.04f}", n, somestr, f);
117
118 /* Clean up. */
119 sir_cleanup();
120
121 return 0;
122}

124int report_error(void) {
